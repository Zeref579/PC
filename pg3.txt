

#include <stdio.h>
#include <omp.h>
int fib(int n) {
if (n < 2) return n;
int x, y;
#pragma omp task shared(x) if(n > 10)
x = fib(n - 1);
#pragma omp task shared(y) if(n > 10)
y = fib(n - 2);
#pragma omp taskwait
return x + y;
}
int main() {
int n;
printf("Enter how many Fibonacci numbers to print: ");
scanf("%d", &n);
#pragma omp parallel
{
#pragma omp single
{
for (int i = 0; i < n; i++) {
int result = fib(i);
printf("fib(%d) = %d\n", i, result);
}
}
}
return 0;
}
